## Design Conventions

| Convention | Rule |
|---|---|
| Primary Keys | `UUID` generated by `uuid_generate_v4()` |
| Timestamps | `created_at TIMESTAMPTZ DEFAULT now()`, `updated_at TIMESTAMPTZ` auto-managed |
| Soft Deletes | `is_active BOOLEAN DEFAULT TRUE`, `deleted_at TIMESTAMPTZ NULL` |
| Money Fields | `DECIMAL(14,2)` — never FLOAT |
| Enums | PostgreSQL `CREATE TYPE` for status fields |
| Audit Log | Append-only; DB trigger prevents UPDATE/DELETE |

## 1. core App

### 1.1 users
```sql
CREATE TABLE users (
    id              UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    email           VARCHAR(255) UNIQUE NOT NULL,
    password_hash   TEXT NOT NULL,                         -- argon2id
    full_name       VARCHAR(255) NOT NULL,
    phone           VARCHAR(20),
    role            role_enum NOT NULL,
    is_active       BOOLEAN DEFAULT TRUE,
    is_locked       BOOLEAN DEFAULT FALSE,
    locked_until    TIMESTAMPTZ,
    last_login      TIMESTAMPTZ,
    created_at      TIMESTAMPTZ DEFAULT now(),
    updated_at      TIMESTAMPTZ DEFAULT now()
);

CREATE TYPE role_enum AS ENUM (
    'super_admin', 'farm_manager', 'accountant',
    'field_worker', 'vet', 'viewer'
);
```

### 1.2 audit_log
```sql
CREATE TABLE audit_log (
    id              UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id         UUID REFERENCES users(id),
    action          VARCHAR(50) NOT NULL,                  -- CREATE, UPDATE, DELETE, LOGIN, LOGOUT
    resource_type   VARCHAR(100) NOT NULL,                 -- e.g. 'cattle', 'milk_collection'
    resource_id     UUID,
    old_values      JSONB,
    new_values      JSONB,
    ip_address      INET,
    user_agent      TEXT,
    created_at      TIMESTAMPTZ DEFAULT now()
    -- NO updated_at — append-only
);

-- Immutability trigger
CREATE OR REPLACE FUNCTION prevent_audit_log_mutation()
RETURNS TRIGGER AS $$
BEGIN
    RAISE EXCEPTION 'audit_log is immutable — INSERT only';
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER audit_log_immutable
BEFORE UPDATE OR DELETE ON audit_log
FOR EACH ROW EXECUTE FUNCTION prevent_audit_log_mutation();
```

### 1.3 login_attempts

```sql
CREATE TABLE login_attempts (
```

## 2. cattle App

### 2.1 cattle
```sql
CREATE TYPE cattle_status AS ENUM ('Active', 'Lactating', 'Dry', 'Pregnant', 'Sold', 'Deceased');
CREATE TYPE cattle_breed  AS ENUM ('Holstein', 'Jersey', 'Gir', 'Sahiwal', 'Murrah', 'Crossbreed', 'Other');

CREATE TABLE cattle (
    id              UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    tag_number      VARCHAR(50) UNIQUE NOT NULL,
    name            VARCHAR(100),
    breed           cattle_breed NOT NULL,
    date_of_birth   DATE NOT NULL,
    status          cattle_status NOT NULL DEFAULT 'Active',
    purchase_date   DATE,
    purchase_price  DECIMAL(14,2),

    deleted_at      TIMESTAMPTZ,
    created_by      UUID REFERENCES users(id),
    created_at      TIMESTAMPTZ DEFAULT now(),
    updated_at      TIMESTAMPTZ DEFAULT now()

CREATE INDEX idx_cattle_status ON cattle(status) WHERE is_active = TRUE;
```

### 2.4 breeding_records
```sql
CREATE TYPE breeding_method AS ENUM ('Natural', 'AI');
CREATE TYPE calving_status  AS ENUM ('Pending', 'Successful', 'Failed', 'Stillbirth');

CREATE TABLE breeding_records (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),

    bull_id             UUID REFERENCES cattle(id),       -- NULL for AI

    calf_tag_number     VARCHAR(50),

    created_at          TIMESTAMPTZ DEFAULT now(),
    updated_at          TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_breeding_expected_calving ON breeding_records(expected_calving)
    WHERE calving_status = 'Pending' OR calving_status IS NULL;
```

## 3. milk App

### 3.1 milk_collections
```sql
CREATE TYPE shift_enum  AS ENUM ('Morning', 'Evening');
CREATE TYPE quality_grade AS ENUM ('A', 'B', 'C', 'Rejected');

CREATE TABLE milk_collections (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    cattle_id           UUID NOT NULL REFERENCES cattle(id),
    field_worker_id     UUID NOT NULL REFERENCES users(id),
    collection_date     DATE NOT NULL,
    shift               shift_enum NOT NULL,
    quantity_litres     DECIMAL(8,2) NOT NULL CHECK (quantity_litres > 0),
    fat_percentage      DECIMAL(4,2) CHECK (fat_percentage BETWEEN 0 AND 10),
    snf_percentage      DECIMAL(4,2) CHECK (snf_percentage BETWEEN 0 AND 12),
    quality_grade       quality_grade,
    temperature_celsius DECIMAL(4,1),
    is_synced           BOOLEAN DEFAULT TRUE,             -- FALSE = offline entry
    notes               TEXT,
    created_at          TIMESTAMPTZ DEFAULT now(),
    updated_at          TIMESTAMPTZ DEFAULT now(),

    CONSTRAINT uq_milk_collection UNIQUE (cattle_id, collection_date, shift)
);
CREATE INDEX idx_milk_date ON milk_collections(collection_date);
CREATE INDEX idx_milk_cattle ON milk_collections(cattle_id);
```

### 3.2 yield_alerts

```sql
CREATE TYPE alert_status AS ENUM ('Open', 'Acknowledged', 'Resolved');

CREATE TABLE yield_alerts (
```

### 5.4 invoices
```sql
CREATE TYPE invoice_status AS ENUM ('Draft','Issued','Paid','PartiallyPaid','Overdue','Cancelled');

CREATE TABLE invoices (
    id              UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    invoice_number  VARCHAR(30) UNIQUE NOT NULL,          -- INV-YYYY-MMDD-NNNN
    order_id        UUID NOT NULL REFERENCES sales_orders(id),
    customer_id     UUID NOT NULL REFERENCES customers(id),
    invoice_date    DATE NOT NULL DEFAULT CURRENT_DATE,
    due_date        DATE NOT NULL,
    subtotal        DECIMAL(14,2) NOT NULL,
    hsn_code        CHAR(4) DEFAULT '0402',
    cgst_rate       DECIMAL(5,2) DEFAULT 0,
    sgst_rate       DECIMAL(5,2) DEFAULT 0,
    igst_rate       DECIMAL(5,2) DEFAULT 0,
    cgst_amount     DECIMAL(14,2) DEFAULT 0,
    sgst_amount     DECIMAL(14,2) DEFAULT 0,
    igst_amount     DECIMAL(14,2) DEFAULT 0,
    total_amount    DECIMAL(14,2) NOT NULL,
    status          invoice_status DEFAULT 'Draft',
    notes           TEXT,
    created_by      UUID REFERENCES users(id),
    created_at      TIMESTAMPTZ DEFAULT now(),
    updated_at      TIMESTAMPTZ DEFAULT now(),

    -- Tax exclusivity: intra-state uses CGST+SGST, inter-state uses IGST only
    CONSTRAINT chk_tax_exclusivity CHECK (
        NOT (igst_amount > 0 AND (cgst_amount > 0 OR sgst_amount > 0))
    )
);
```

### 6.4 payroll_lines
```sql
CREATE TABLE payroll_lines (

    run_id          UUID NOT NULL REFERENCES payroll_runs(id),
    employee_id     UUID NOT NULL REFERENCES employees(id),
    present_days    SMALLINT NOT NULL DEFAULT 0,
    working_days    SMALLINT NOT NULL,
    basic_salary    DECIMAL(14,2) NOT NULL,
    earned_basic    DECIMAL(14,2) NOT NULL,               -- basic * (present/working)
    pf_employee     DECIMAL(14,2) DEFAULT 0,              -- 12% of earned_basic
    pf_employer     DECIMAL(14,2) DEFAULT 0,              -- 12% of earned_basic
    esi_employee    DECIMAL(14,2) DEFAULT 0,              -- 0.75% of gross
    esi_employer    DECIMAL(14,2) DEFAULT 0,              -- 3.25% of gross
    other_deductions DECIMAL(14,2) DEFAULT 0,
    net_pay         DECIMAL(14,2) NOT NULL
                        CHECK (net_pay >= 0),
    created_at      TIMESTAMPTZ DEFAULT now()
);
```

### 7.3 journal_lines
```sql
CREATE TABLE journal_lines (
    id              UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    entry_id        UUID NOT NULL REFERENCES journal_entries(id),
    account_id      UUID NOT NULL REFERENCES accounts(id),
    debit           DECIMAL(14,2) DEFAULT 0,
    credit          DECIMAL(14,2) DEFAULT 0,
    notes           TEXT,
    CHECK (debit >= 0 AND credit >= 0),
    CHECK (NOT (debit > 0 AND credit > 0))               -- one side only per line
);
-- Double-entry balance enforced in application service layer before commit
```

## 8. Indexes Summary

| Table | Index | Purpose |
|---|---|---|
| users | email UNIQUE | Login lookup |
| cattle | tag_number UNIQUE | Tag scan lookup |
| cattle | (status) WHERE is_active | Active cattle list |
| milk_collections | UNIQUE(cattle_id, date, shift) | Duplicate prevention (FR-M-01) |
| milk_collections | (collection_date) | Daily summary queries |
| vaccinations | (next_due_date) | Upcoming vaccine alerts |
| breeding_records | (expected_calving) WHERE pending | Calving alerts |
| login_attempts | (user_id, attempted_at) | Lockout counting |
| audit_log | (resource_type, resource_id) | Resource history lookup |
| payroll_runs | UNIQUE(month, year) | One payroll run per period |
